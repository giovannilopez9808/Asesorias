\section*{Tarea Clase 7}
\subsection*{Obtenga las soluciones de las siguientes ecuaciones:}
\begin{enumerate}
    \item\begin{equation*}
        \left\lbrace \begin{matrix}
             4x+3y=10 \\
             5x-6y=-7
        \end{matrix} \right.
    \end{equation*}
    \begin{align*}
        x&=1\\
        y&=2
    \end{align*}
    \item\begin{equation*}
        \left\lbrace \begin{matrix}
           2x+y=-10 \\
           x-3y=2  
        \end{matrix} \right.
    \end{equation*}
    \begin{align*}
        x&=-4\\
        y&=-2
    \end{align*}
    \item\begin{equation*}
        \left\lbrace \begin{matrix}
             6r-5t=-11\\
             7t-8r=15
        \end{matrix} \right.
    \end{equation*}\begin{align*}
        r&=-1\\
        t&=1
    \end{align*}
    \item\begin{equation*}
        \left\lbrace \begin{matrix}
             12u-16v=24\\
             3u-4v=6
        \end{matrix} \right.
    \end{equation*}
    Este problema tiene soluciones infinitas, ya que resulta por cualquier método que 0=0
    \item\begin{equation*}
        \left\lbrace \begin{matrix}
             2x+y=9\\
             8x+4y=36
        \end{matrix} \right.
    \end{equation*}
    Este problema tiene soluciones infinitas
    \item\begin{equation*}
        \left\lbrace \begin{matrix}
             4p-3q=-2\\
             20p-15q=-1
        \end{matrix} \right.
    \end{equation*}
    Este problema no tiene solución, ya que queda 0=\#, donde \# es cualquier numerp
    \item\begin{equation*}
        \left\lbrace \begin{matrix}
             6x-7y=5\\
             8x-9y=7
        \end{matrix} \right.
    \end{equation*}\begin{align*}
        x&=2\\
        y&=1
    \end{align*}
    \item\begin{equation*}
        \left\lbrace \begin{matrix}
             x+5y=5\\
             3x-5y=3
        \end{matrix} \right.
    \end{equation*}\begin{align*}
        x&=-1\\
        y&=6/5
    \end{align*}
    \item\begin{equation*}
        \left\lbrace \begin{matrix}
             y=-2x+1\\
             4x+2y=3
        \end{matrix} \right.
    \end{equation*}
    Este problema no tiene solución.
    \item\begin{equation*}
        \left\lbrace \begin{matrix}
             2y-3x=1\\
             -4y+6x=-2
        \end{matrix} \right.
    \end{equation*}
    Este problema contiene soluciones infinitas
    \item\begin{equation*}
        \left\lbrace \begin{matrix}
             6x-5y=-3\\
             3x+2y=12
        \end{matrix} \right.
    \end{equation*}
    \begin{align*}
        x&=2\\
        y&=3
    \end{align*}
\end{enumerate}
