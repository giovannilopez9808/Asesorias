\section*{Tarea Clase 5}
\subsection*{Realiza las siguientes operaciones:}
\begin{enumerate}
\item \begin{equation*}
\frac{5}{x}-\frac{3x+2}{x+1}+\frac{2}{x^2+x}
\end{equation*}
\begin{align*}
    &\frac{5(x+1)+x(3x+2)+2}{x(x+1)}\\
    &\frac{5x+5+3x^2+2x+2}{x(x+1)}\\
    &\frac{3x^2+7x+7}{x(x+1)}
\end{align*}
\item \begin{equation*}
\frac{3x}{2x-3}+\frac{1}{4x^2-9}-\frac{6x^2}{3+2x}
\end{equation*}
\begin{align*}
    &\frac{3x(2x+3)+1-6x^2(2x-3)}{(2x+3)(2x-3)}\\
    &\frac{6x^2+9x+1-12x^2+18x^2}{(2x+3)(2x-3)}\\
    &\frac{12x^2+9x+1}{(2x-3)(2x+3)}
\end{align*}
\item \begin{equation*}
\frac{x+2}{x^2+x+1}-\frac{5-2x}{x^2-4}
\end{equation*}
\begin{align*}
    \frac{(x^2-4)(x+2)-(5-2x)(x^2+x+1)}{(x^2+x+1)(x-2)(x+2)}
\end{align*}
\item \begin{equation*}
    \left(\frac{a-b}{2ab}\right) \left(\frac{a-b}{-2ab} \right)
\end{equation*}
\begin{align*}
    &\frac{a^2-2ab+b^2}{-4a^2b^2}
\end{align*}
\item \begin{equation*}
    \left(\frac{1-2a+a^2}{4m^2-4m+1}\right) \cdot \left( \frac{(2m-1)^2}{a-1} \right)
\end{equation*}
\begin{align*}
    &\left( \frac{(a-1)^2}{(2m-1)^2}\right)\left(\frac{(2m-1)^2}{a-1} \right) \\
    &a-1
\end{align*}
\item \begin{equation*}
    \frac{x^2-4}{x^2+2x-3} \div \frac{x^2-x-2}{x-1}
\end{equation*}
\begin{align*}
    &\frac{(x-2)(x+2)}{(x+3)(x-1)}\frac{x-1}{(x+1)(x-2)}\\
    &\frac{(x+2)}{(x+3)(x+1)}
\end{align*}
\item \begin{equation*}
    \frac{x^2-3x}{x^2+2x+1} \cdot \frac{x^2-1}{x^2-9}
\end{equation*}
\begin{align*}
    &\frac{x(x-3)}{(x+1)^2}\frac{(x-1)(x+1)}{(x-3)(x+3)}\\
    &\frac{x(x-1)}{(x+1)(x+3)}
\end{align*}
\item \begin{equation*}
    \frac{x-1}{2x^2+4x+2} \cdot \frac{x+1}{x-1}
\end{equation*}
\begin{align*}
    & \frac{(x-1)(x+1)}{2(x+1)^2(x-1)}\\
    &\frac{1}{2(x+1)}
\end{align*}
\item \begin{equation*}
    \frac{x^3-x}{3x-6} \div \frac{5x+5}{2x-4}
\end{equation*}
\begin{align*}
    &\frac{(x)(x-1)(x+1)(2)(x-2)}{(5)(x+1)(3)(x-2)}\\
    &\frac{2x(x-1)}{15}
\end{align*}
\end{enumerate}
